  sheet.clear(); // يمسح البيانات القديمة
  
  // يجيب أسماء الأعمدة من أول عنصر في JSON
  const headers = Object.keys(json[0]);
  sheet.appendRow(headers);
  
  // يضيف كل سجل في صف مستقل

    sheet.appendRow(row);
  });
